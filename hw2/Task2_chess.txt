x1 = int(input('Введи номер столбца начальной клетки '))
y1 = int(input('Введи номер строки начальной клетки '))
x2 = int(input('Введи номер столбца конечной клетки '))
y2 = int(input('Введи номер строки конечной клетки '))

if (x1 == y2) and (x2 == y2):
    print('Ладья не может совершить такой ход')
    print('Конь не может совершить такой ход')
    print('Слон не может совершить такой ход')
else:
    if (x1 == x2) or (y1 == y2):
        print('Ладья может совершить такой ход')
    else:
        print('Ладья не может совершить такой ход')


    if ((y1 == y2 + 2) or (y1 == y2 - 2)) and ((x1 == x2 + 1) or (x2 == x2 - 1)):
        print('Конь может совершить такой ход')
    else:
        print('Конь не может совершить такой ход')

    if abs(x1 - x2) == abs(y1 - y2):
        print('Слон может совершить такой ход')
    else:
        print('Слон не может совершить такой ход')
