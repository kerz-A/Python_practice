x = float(input('введи положительное вещественное число x: '))
y = float(input('введи положительное вещественное число y: '))
z = float(input('введи положительное вещественное число x: '))
exp = x**2*(y**0.5 + z**0.5)*(1/((x**6+y**0.5)**0.5)) - (3*x)**0.5 + z**11
print(f'Вывод программы:  {exp:.3f}')
